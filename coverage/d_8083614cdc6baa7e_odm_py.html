76def check_version(timeout: int, max_retries: int) -> None: 

77 sess = init_session(max_retries) 

78 # noinspection PyBroadException 

79 try: 

80 res = sess.get(TAGS_ENDPOINT, timeout=timeout) 

81 res.raise_for_status() 

82 curr_version = res.json()[0].get("name", "") 

83 if curr_version and curr_version != __version__: 

84 logger.warning( 

85 f"⚠️ A new version {curr_version} is available at {REPOSITORY_URL}." 

86 ) 

87 except: # noqa: E722, pylint: disable=bare-except 

88 pass

266def extract_loan_file( 

267 libby_client: LibbyClient, selected_loan: Dict, args: argparse.Namespace 

268) -> Optional[Path]: 

269 """ 

270 Extracts the ODM / ACSM / EPUB(open) file 

271 

272 :param libby_client: 

273 :param selected_loan: 

274 :param args: 

275 :return: The path to the ODM file 

276 """ 

277 try: 

278 format_id = LibbyClient.get_loan_format(selected_loan) 

279 except ValueError as err: 

280 err_msg = str(err) 

281 if "kindle" in str(err): 

282 logger.error( 

283 "You may have already sent the loan to your Kindle device: %s", 

284 colored(err_msg, "red"), 

285 ) 

286 else: 

287 logger.error(colored(err_msg, "red")) 

288 return None 

289 

290 file_ext = "odm" 

291 file_name = f'{selected_loan["title"]} {selected_loan["id"]}' 

292 loan_file_path = Path( 

293 args.download_dir, f"{slugify(file_name, allow_unicode=True)}.{file_ext}" 

294 ) 

295 if ( 

296 args.libby_direct 

297 and libby_client.has_format(selected_loan, LibbyFormats.EBookOverdrive) 

298 and not ( 

299 # don't do direct downloads for PDF loans because these turn out badly 

300 libby_client.has_format(selected_loan, LibbyFormats.EBookPDFAdobe) 

301 or libby_client.has_format(selected_loan, LibbyFormats.EBookPDFOpen) 

302 ) 

303 ): 

304 format_id = LibbyFormats.EBookOverdrive 

305 

306 openbook: Dict = {} 

307 rosters: List[Dict] = [] 

308 # pre-extract openbook first so that we can use it to create the book folder 

309 # with the creator names (needed to place the cover.jpg download) 

310 if format_id in (LibbyFormats.EBookOverdrive, LibbyFormats.MagazineOverDrive): 

311 _, openbook, rosters = libby_client.process_ebook(selected_loan) 

312 

313 cover_path = None 

314 if format_id in ( 

315 LibbyFormats.EBookEPubAdobe, 

316 LibbyFormats.EBookEPubOpen, 

317 LibbyFormats.EBookOverdrive, 

318 LibbyFormats.MagazineOverDrive, 

319 LibbyFormats.EBookPDFAdobe, 

320 LibbyFormats.EBookPDFOpen, 

321 ): 

322 file_ext = ( 

323 "acsm" 

324 if format_id in (LibbyFormats.EBookEPubAdobe, LibbyFormats.EBookPDFAdobe) 

325 else "pdf" 

326 if format_id == LibbyFormats.EBookPDFOpen 

327 else "epub" 

328 ) 

329 book_folder, book_file_name = generate_names( 

330 title=selected_loan["title"], 

331 series=selected_loan.get("series") or "", 

332 authors=extract_authors_from_openbook(openbook) 

333 or ( 

334 [selected_loan["firstCreatorName"]] 

335 if selected_loan.get("firstCreatorName") 

336 else [] 

337 ), # for open-epub 

338 edition=selected_loan.get("edition") or "", 

339 title_id=selected_loan["id"], 

340 args=args, 

341 logger=logger, 

342 ) 

343 loan_file_path = book_file_name.with_suffix(f".{file_ext}") 

344 if ( 

345 format_id 

346 in ( 

347 LibbyFormats.EBookOverdrive, 

348 LibbyFormats.MagazineOverDrive, 

349 ) 

350 and not loan_file_path.exists() 

351 ): 

352 # we need the cover for embedding 

353 cover_path, _ = generate_cover( 

354 book_folder=book_folder, 

355 cover_url=get_best_cover_url(selected_loan), 

356 session=init_session(args.retries), 

357 timeout=args.timeout, 

358 logger=logger, 

359 force_square=False, 

360 ) 

361 

362 # don't re-download odm if it already exists so that we don't 

363 # needlessly use up the fulfillment limits 

364 if not loan_file_path.exists(): 

365 if format_id in (LibbyFormats.EBookOverdrive, LibbyFormats.MagazineOverDrive): 

366 process_ebook_loan( 

367 loan=selected_loan, 

368 cover_path=cover_path, 

369 openbook=openbook, 

370 rosters=rosters, 

371 libby_client=libby_client, 

372 args=args, 

373 logger=logger, 

374 ) 

375 else: 

376 # formats: odm, acsm, open-epub, open-pdf 

377 try: 

378 odm_res_content = libby_client.fulfill_loan_file( 

379 selected_loan["id"], selected_loan["cardId"], format_id 

380 ) 

381 with loan_file_path.open("wb") as f: 

382 f.write(odm_res_content) 

383 logger.info( 

384 'Downloaded %s to "%s"', 

385 file_ext, 

386 colored(str(loan_file_path), "magenta"), 

387 ) 

388 except ClientError as ce: 

389 if ce.http_status == 400 and libby_client.is_downloadable_ebook_loan( 

390 selected_loan 

391 ): 

392 logger.error( 

393 "%s %s", 

394 colored( 

395 f"Unable to download {file_ext}.", "red", attrs=["bold"] 

396 ), 

397 colored("You may have sent the loan to a Kindle.", "red"), 

398 ) 

399 return None 

400 raise 

401 else: 

402 logger.info( 

403 "Already downloaded %s file %s", 

404 file_ext, 

405 colored(str(loan_file_path), "magenta"), 

406 ) 

407 

408 if cover_path and cover_path.exists() and not args.always_keep_cover: 

409 # clean up 

410 cover_path.unlink() 

411 

412 return loan_file_path
